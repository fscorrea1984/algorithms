\begin{algorithm}
  \caption{mergesort}
  \begin{algorithmic}[1]
    \Function{mergesort}{$A, p, r$}
    \If{$p \geq r$}\Comment{Zero or one element?}
    \State return
    \EndIf
    \State $q = \lfloor (p + r)/2 \rfloor$ \Comment{midpoint of $A[p : r]$}
    \State mergesort(A, p, q) \Comment{recursively sort $A[p : q]$}
    \State mergesort(A, q + 1, r) \Comment{recursively sort $A[q + 1 : r]$} \\
    \Comment{ merge $A[p : q]$ and $A[q + 1 : r]$ into $A[p : r]$} \\
    \State merge(A, p, q, r)
    \EndFunction
  \end{algorithmic}
\end{algorithm}
