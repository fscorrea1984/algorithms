\begin{algorithm}
  \caption{insertionsort}
  \begin{algorithmic}[1]
    \Function{insertionsort}{$A,n$}
    \For{$i = 2$ to $n$}
    \State $key = A[i]$ \\
    \Comment{Insert $A[i]$ into the sorted subarray $A[1 : i - 1]$}
    \State $j = i - 1$
    \While{$j > 0$ and $A[j] > key$}
    \State $A[j + 1] = A[j]$
    \State $j = j - 1$
    \EndWhile
    \State $A[j + 1] = key$
    \EndFor
    \EndFunction
  \end{algorithmic}
\end{algorithm}
